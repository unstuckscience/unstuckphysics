  <section class="solution-wrapper">

    <button type="button" class="btn btn--primary solution-toggle" aria-expanded="false" aria-controls="solution-content">

      Reveal Solution

    </button>

    <div id="solution-content" class="solution-hide" aria-hidden="true">

      <h3>Solution</h3> {{ page.solution_content }} </div>

  </section>


<script>

  document.addEventListener('DOMContentLoaded', function() {

    const toggleButton = document.querySelector('.solution-toggle');

    const solutionContent = document.getElementById('solution-content');


    if (toggleButton && solutionContent) {

      toggleButton.addEventListener('click', function() {

        const isHidden = solutionContent.classList.contains('solution-hide');


        if (isHidden) {

          solutionContent.classList.remove('solution-hide');

          solutionContent.classList.add('solution-show');

          this.setAttribute('aria-expanded', 'true');

          solutionContent.setAttribute('aria-hidden', 'false');

          this.textContent = 'Hide Solution'; // Change button text

        } else {

          solutionContent.classList.remove('solution-show');

          solutionContent.classList.add('solution-hide');

          this.setAttribute('aria-expanded', 'false');

          solutionContent.setAttribute('aria-hidden', 'true');

          this.textContent = 'Reveal Solution'; // Change button text

        }

      });

    }

  });

</script>
